################################## 1. Initial Setup ####################################

### A. OS Installation (Custom Partitioning)
# To protect your AI models from future OS reinstallations, separate your root and home drives:
# 1. Boot the Fedora 43 Server installer.
# 2. Go to "Installation Destination" -> Select your 2TB drive.
# 3. Under Storage Configuration, select "Custom" -> Click "Done".
# 4. Click the blue link to "create them automatically" (this builds your boot/EFI).
# 5. Select the `/` (root) partition. Change "Desired Capacity" to 100 GiB.
#    (100GB gives plenty of room for Docker images and the AMD toolboxes).
# 6. Select the `/home` partition. Leave "Desired Capacity" blank to use all remaining space (~1.8 TiB).
# 7. Click "Done" and accept changes to build the isolated layout.

### B. Post-Install Updates
# Update System
sudo dnf update -y

### C. Verifying your setup
uname -a

### Verify Kernel Version (Must be 6.18.4 or newer)
uname -r

### Verify Firmware (Do NOT use 20251125, It breaks ROCm support on Strix Halo)
rpm -qa | grep linux-firmware

################################## 2. Kernel Config ####################################
### Install nano
sudo dnf install nano vim bash-completion -y

### Install the ROCm utilities
sudo dnf install rocminfo rocm-smi -y

### Edit Grub
sudo nano /etc/default/grub

### Append to GRUB_CMDLINE_LINUX line:
GRUB_TIMEOUT=5
GRUB_DISTRIBUTOR="$(sed 's, release .*$,,g' /etc/system-release)"
GRUB_DEFAULT=saved
GRUB_DISABLE_SUBMENU=true
GRUB_TERMINAL_OUTPUT="console"
GRUB_CMDLINE_LINUX="rd.lvm.lv=fedora_<hostname>/root rhgb quiet iommu=pt amdgpu.gttsize=126976 ttm.pages_limit=32505856"
GRUB_DISABLE_RECOVERY="true"
GRUB_ENABLE_BLSCFG=true

### Apply Grub Changes
sudo grub2-mkconfig -o /boot/grub2/grub.cfg

sudo reboot

### Grub edit
cat /etc/default/grub

## Output should be something like below
# GRUB_CMDLINE_LINUX="r.luks.uuid=luks2ca3232b-d7f6-4409-83ed-a73544f1b27a rhgb quite amd_iommu=off amdgpu.gttsize=131072 ttm.pages_limit=335544321"


################################## 3. Toolbox Setup ####################################
### Install Toolbox (If not present on Server install)

toolbox enter llama-rocm-7.2

### Ollama (Service Setup)
# Install Script
curl -fsSL https://ollama.com/install.sh | sh

# Configure Storage to /home (Override default /usr/share)
sudo mkdir -p /home/<username>/ai_models

# 1. Give the ollama user ownership of the model folder
sudo chown -R ollama:ollama /home/<username>/ai_models

# 2. Set permissions so the 'ollama' group can write and the user can 'enter' the path
sudo chmod 755 /home/<username>
sudo chmod -R 775 /home/<username>/ai_models

# 3. Update the Service to point to the new folder
sudo systemctl edit ollama.service

# In the editor, add the following lines at the top:
# [Service]
# Environment="OLLAMA_MODELS=/home/<username>/ai_models"

# 4. Reload and Restart the service
sudo systemctl daemon-reload
sudo systemctl restart ollama

### Llama.cpp (Native Vulkan Build for 3060 Ti & AMD)
# Install Vulkan Build Dependencies
sudo dnf install git make gcc-c++ cmake vulkan-headers vulkan-loader-devel glslc -y

# Clone Repository
cd /home/<username>
git clone https://github.com/ggerganov/llama.cpp
cd llama.cpp

# Configure and Build with Vulkan Support
cmake -B build -DGGML_VULKAN=ON
cmake --build build --config Release -j $(nproc)

# Usage Note:
# Use ./build/bin/llama-cli (in this folder) to run the natively compiled version.
# Use the 'toolbox' commands from the previous section to use the AMD Strix Halo GPU exclusively.

### TailScale
# Add Repo and Install
sudo dnf config-manager addrepo --from-repofile=https://pkgs.tailscale.com/stable/fedora/tailscale.repo
sudo dnf install tailscale -y

# Start Service
sudo systemctl enable --now tailscaled
sudo tailscale up

	# Number of CPU threads to use during generation (default: -1)

	# Max number of layers to store in VRAM (auto, all, or exact number)

	# Number of server slots (default: -1, auto)

	# Set max cache size in MiB (default: 8192)
    --cache-ram -1 \
	# Set Flash Attention use (default: 'auto')
    --flash-attn on \
	# Memory-map model (default: enabled)
    --no-mmap \
	# IP address to listen (default: 127.0.0.1)
    --host 0.0.0.0 \
	# Port to listen (default: 8080)
    --port 8080 \
	# Set verbosity level to infinity
    --verbose \
	# Physical maximum batch size (default: 512)
    --ubatch-size 512 \
	# Logical maximum batch size (default: 2048)
    --batch-size 2048 \
	#  KV cache data type for K (f32, f16, bf16, q8_0, etc. Default: f16)
    -ctk q8_0 \
	# KV cache data type for V (f32, f16, bf16, q8_0, etc. Default: f16)
    -ctv q8_0
