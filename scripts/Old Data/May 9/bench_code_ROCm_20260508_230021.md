```r

library(dplyr)

# Load shapefiles
load_properties <- function(properties_path) {
  tryCatch(
    expr = st_read(properties_path),
    error = function(e) stop(paste("Failed to load properties shapefile:", e))
  )
}

load_golf_courses <- function(golf_courses_path) {
  tryCatch(
    expr = st_read(golf_courses_path),
    error = function(e) stop(paste("Failed to load golf courses shapefile:", e))
  )
}

# Perform spatial join and calculate distance
calculate_distance_to_golf_course <- function(properties, golf_courses) {
  properties_with_distance <- properties %>%
    st_intersection(golf_courses) %>%
    mutate(dist_to_golf_m = ifelse(!is.na(st_distance)), 
           dist_to_golf_m = st_distance / 1000)) # convert to meters
  
  # For properties not intersecting with golf courses, calculate distance
  non_intersecting_properties <- properties[!properties$geometry %>% st_intersects(golf_courses), ] %>%
    st_distance(golf_courses) %>%
    apply(2, min)
  
  if (nrow(properties_with_distance) < nrow(properties)) {
    dist_to_golf_m <- c(properties_with_distance$dist_to_golf_m,
                        rep(NA, nrow(properties) - nrow(properties_with_distance)))
    
    properties_updated <- properties
    properties_updated$dist_to_golf_m <- dist_to_golf_m
    
  } else {
    properties_updated <- properties_with_distance
  }
  
  return(properties_updated)
}

# Main function with example usage
main <- function() {
  # Specify shapefile paths
  properties_path <- "path/to/properties.shp"
  golf_courses_path <- "path/to/golf_courses.shp"

  # Load shapefiles
  properties <- load_properties(properties_path)
  golf_courses <- load_golf_courses(golf_courses_path)

  # Perform spatial join and calculate distance
  updated_properties <- properties %>%
    st_transform(CRS = golf_courses) %>% 
    calculate_distance_to_golf_course(golf_courses)

  return(updated_properties)
}

# Run the main function
updated_df <- main()
print(updated_df)
```

**Note:**

* Make sure to replace `"path/to/properties.shp"` and `"path/to/golf_courses.shp"` with your actual shapefile paths.
* The `calculate_distance_to_golf_course` function assumes that both shapefiles are in the same CRS (Coordinate Reference System). If not, it transforms the properties shapefile to match the golf courses' CRS. 
* This script calculates distances in meters and stores them as a new column called 'dist_to_golf_m'. 

The updated dataframe `updated_df` will contain all columns from the original `properties.shp`, plus an additional column for distance to nearest golf course boundary, which can be used for further analysis or processing.
